## 📝 使用示例

### 从剧本生成分镜提示词

```python
# 示例：使用模板库生成提示词

from pathlib import Path
import yaml

class StoryboardPromptGenerator:
    def __init__(self, template_path):
        self.templates = yaml.safe_load(Path(template_path).read_text())
    
    def generate_shot_prompt(self, shot_data):
        """根据五维架构生成提示词"""
        prompt_parts = []
        
        # 主体维度
        if shot_data.get('subject'):
            prompt_parts.append(shot_data['subject'])
        
        # 动作维度
        if shot_data.get('action'):
            prompt_parts.append(shot_data['action'])
        
        # 环境维度
        if shot_data.get('environment'):
            prompt_parts.append(shot_data['environment'])
        
        # 情感维度
        if shot_data.get('emotion'):
            prompt_parts.append(shot_data['emotion'])
        
        # 技术维度
        if shot_data.get('technical'):
            tech = shot_data['technical']
            shot_type = tech.get('type', 'medium shot')
            angle = tech.get('angle', 'eye level')
            movement = tech.get('movement', 'static')
            prompt_parts.append(f"{shot_type}, {angle} angle, {movement} camera")
        
        # 风格后缀
        style_suffix = self.templates.get('style_suffix', 'anime style, high quality')
        prompt_parts.append(style_suffix)
        
        return ", ".join(prompt_parts)

# 使用示例
generator = StoryboardPromptGenerator('prompts/templates/anime_style.yaml')
shot_prompt = generator.generate_shot_prompt({
    'subject': 'A girl with short black hair in red hoodie',
    'action': 'standing at the edge of rooftop',
    'environment': 'urban city skyline at sunset',
    'emotion': 'peaceful melancholic atmosphere',
    'technical': {
        'type': 'wide establishing shot',
        'angle': 'eye level',
        'movement': 'slow push in'
    }
})
print(shot_prompt)
# 输出: "A girl with short black hair in red hoodie, standing at the edge of rooftop, 
#        urban city skyline at sunset, peaceful melancholic atmosphere, 
#        wide establishing shot, eye level angle, slow push in camera, 
#        anime style, high quality"
```
